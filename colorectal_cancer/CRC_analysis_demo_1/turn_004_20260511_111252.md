## Data Levels: Cell vs Spot

**Cell-Level Data** (single-cell resolution):
- Each row = one cell
- High resolution (~10μm)
- Supports: clustering, cell type annotation, niche detection
- Detection: `slice.is_cell_level`

**Spot-Level Data** (Visium, spatial transcriptomics):
- Each row = one tissue spot (~50-100μm diameter)
- Contains MIXTURE of multiple cells
- Supports: deconvolution (celltype proportions), regional analysis
- Detection: `slice.is_spot_level`
- **Special fields:**
  - `adata.uns['spot_shape']`: 'circle' or 'square'
  - `adata.uns['spot_diameter']`: Diameter in micrometers
  - `adata.obsm['deconv_weights']`: DataFrame (spots × celltypes) with proportions (May be missing if not deconvolved)
- **Virtual celltype:** `adata.obs['celltype']` shows DOMINANT type (may be misleading)
- **CANNOT use cell-level clustering** - spots are not cells!

def get_vector(adata, gene):
    """Return dense 1D expression vector for one gene."""
    x = adata[:, gene].X
    if sparse.issparse(x):
        x = x.toarray()
    return np.asarray(x).ravel()


def mean_expr_for_genes(adata, obs_mask, genes):
    """
    Efficiently compute mean expression and pct expressing for a list of genes
    in a selected spot/cell group.
    """
    genes = [g for g in genes if g in adata.var_names]
    if len(genes) == 0:
        return pd.DataFrame(columns=["gene", "mean_expr", "pct_expr"])

    sub = adata[obs_mask, genes].X
    if sparse.issparse(sub):
        means = np.asarray(sub.mean(axis=0)).ravel()
        pct = np.asarray((sub > 0).mean(axis=0)).ravel()
    else:
        sub = np.asarray(sub)
        means = sub.mean(axis=0)
        pct = (sub > 0).mean(axis=0)

    return pd.DataFrame({
        "gene": genes,
        "mean_expr": means,
        "pct_expr": pct
    })


def rank_group_lr_scores(
    adata,
    group_col,
    source_groups,
    target_groups,
    lr_df,
    slice_id,
    patient,
    min_source_spots=10,
    min_target_spots=10,
):
    """
    Score ligand-receptor communication using source ligand mean expression
    and target receptor mean expression.

    Score = sqrt(mean_ligand_source * mean_receptor_target)
    pct_product = pct_ligand_source * pct_receptor_target
    """
    available_genes = set(adata.var_names.astype(str))
    lr = lr_df[
        lr_df["ligand"].isin(available_genes) &
        lr_df["receptor"].isin(available_genes)
    ].copy()

    if lr.empty:
        return pd.DataFrame()

    needed_ligands = sorted(lr["ligand"].unique())
    needed_receptors = sorted(lr["receptor"].unique())

    rows = []

    for src in source_groups:
        src_mask = (adata.obs[group_col].astype(str).values == src)
        n_src = int(src_mask.sum())
        if n_src < min_source_spots:
            print(f"Slice {slice_id}: skipping source {src} because n={n_src} < {min_source_spots}")
            continue

        src_lig = mean_expr_for_genes(adata, src_mask, needed_ligands)
        src_lig = src_lig.rename(columns={
            "gene": "ligand",
            "mean_expr": "ligand_mean_source",
            "pct_expr": "ligand_pct_source"
        })

        for tgt in target_groups:
            tgt_mask = (adata.obs[group_col].astype(str).values == tgt)
            n_tgt = int(tgt_mask.sum())
            if n_tgt < min_target_spots:
                print(f"Slice {slice_id}: skipping target {tgt} because n={n_tgt} < {min_target_spots}")
                continue

            tgt_rec = mean_expr_for_genes(adata, tgt_mask, needed_receptors)
            tgt_rec = tgt_rec.rename(columns={
                "gene": "receptor",
                "mean_expr": "receptor_mean_target",
                "pct_expr": "receptor_pct_target"
            })

            tmp = (
                lr.merge(src_lig, on="ligand", how="inner")
                  .merge(tgt_rec, on="receptor", how="inner")
            )

            if tmp.empty:
                continue

            tmp["slice_id"] = slice_id
            tmp["patient"] = patient
            tmp["source"] = src
            tmp["target"] = tgt
            tmp["n_source"] = n_src
            tmp["n_target"] = n_tgt
            tmp["lr_pair"] = tmp["ligand"] + "→" + tmp["receptor"]

            tmp["comm_score"] = np.sqrt(
                np.maximum(tmp["ligand_mean_source"], 0) *
                np.maximum(tmp["receptor_mean_target"], 0)
            )
            tmp["pct_product"] = tmp["ligand_pct_source"] * tmp["receptor_pct_target"]

            # Directional specificity compared with all non-source / non-target spots.
            non_src_mask = ~src_mask
            non_tgt_mask = ~tgt_mask

            # Only calculate lightweight specificity using already computed genes.
            # These background means help downweight ubiquitous LR expression.
            bg_lig = mean_expr_for_genes(adata, non_src_mask, needed_ligands).rename(columns={
                "gene": "ligand",
                "mean_expr": "ligand_mean_background_source"
            })[["ligand", "ligand_mean_background_source"]]

            bg_rec = mean_expr_for_genes(adata, non_tgt_mask, needed_receptors).rename(columns={
                "gene": "receptor",
                "mean_expr": "receptor_mean_background_target"
            })[["receptor", "receptor_mean_background_target"]]

            tmp = tmp.merge(bg_lig, on="ligand", how="left").merge(bg_rec, on="receptor", how="left")

            eps = 1e-9
            tmp["ligand_log2fc_source"] = np.log2((tmp["ligand_mean_source"] + eps) /
                                                  (tmp["ligand_mean_background_source"] + eps))
            tmp["receptor_log2fc_target"] = np.log2((tmp["receptor_mean_target"] + eps) /
                                                    (tmp["receptor_mean_background_target"] + eps))
            tmp["specificity_score"] = tmp["comm_score"] * np.maximum(tmp["ligand_log2fc_source"], 0) * np.maximum(tmp["receptor_log2fc_target"], 0)

            rows.append(tmp)

    if len(rows) == 0:
        return pd.DataFrame()

    res = pd.concat(rows, ignore_index=True)

    # Ranking: prefer strong expression and target/source specificity.
    # If specificity is all zero, communication score still ranks.
    res["rank_score"] = res["specificity_score"]
    if np.nanmax(res["rank_score"].values) <= 0:
        res["rank_score"] = res["comm_score"] * res["pct_product"]

    res = res.sort_values(
        ["rank_score", "comm_score", "pct_product"],
        ascending=False
    ).reset_index(drop=True)

    return res

def get_vector(adata, gene):
    """Return dense 1D expression vector for one gene."""
    x = adata[:, gene].X
    if sparse.issparse(x):
        x = x.toarray()
    return np.asarray(x).ravel()


def mean_expr_for_genes(adata, obs_mask, genes):
    """
    Efficiently compute mean expression and pct expressing for a list of genes
    in a selected spot/cell group.
    """
    genes = [g for g in genes if g in adata.var_names]
    if len(genes) == 0:
        return pd.DataFrame(columns=["gene", "mean_expr", "pct_expr"])

    sub = adata[obs_mask, genes].X
    if sparse.issparse(sub):
        means = np.asarray(sub.mean(axis=0)).ravel()
        pct = np.asarray((sub > 0).mean(axis=0)).ravel()
    else:
        sub = np.asarray(sub)
        means = sub.mean(axis=0)
        pct = (sub > 0).mean(axis=0)

    return pd.DataFrame({
        "gene": genes,
        "mean_expr": means,
        "pct_expr": pct
    })


def rank_group_lr_scores(
    adata,
    group_col,
    source_groups,
    target_groups,
    lr_df,
    slice_id,
    patient,
    min_source_spots=10,
    min_target_spots=10,
):
    """
    Score ligand-receptor communication using source ligand mean expression
    and target receptor mean expression.

    Score = sqrt(mean_ligand_source * mean_receptor_target)
    pct_product = pct_ligand_source * pct_receptor_target
    """
    available_genes = set(adata.var_names.astype(str))
    lr = lr_df[
        lr_df["ligand"].isin(available_genes) &
        lr_df["receptor"].isin(available_genes)
    ].copy()

    if lr.empty:
        return pd.DataFrame()

    needed_ligands = sorted(lr["ligand"].unique())
    needed_receptors = sorted(lr["receptor"].unique())

    rows = []

    for src in source_groups:
        src_mask = (adata.obs[group_col].astype(str).values == src)
        n_src = int(src_mask.sum())
        if n_src < min_source_spots:
            print(f"Slice {slice_id}: skipping source {src} because n={n_src} < {min_source_spots}")
            continue

        src_lig = mean_expr_for_genes(adata, src_mask, needed_ligands)
        src_lig = src_lig.rename(columns={
            "gene": "ligand",
            "mean_expr": "ligand_mean_source",
            "pct_expr": "ligand_pct_source"
        })

        for tgt in target_groups:
            tgt_mask = (adata.obs[group_col].astype(str).values == tgt)
            n_tgt = int(tgt_mask.sum())
            if n_tgt < min_target_spots:
                print(f"Slice {slice_id}: skipping target {tgt} because n={n_tgt} < {min_target_spots}")
                continue

            tgt_rec = mean_expr_for_genes(adata, tgt_mask, needed_receptors)
            tgt_rec = tgt_rec.rename(columns={
                "gene": "receptor",
                "mean_expr": "receptor_mean_target",
                "pct_expr": "receptor_pct_target"
            })

            tmp = (
                lr.merge(src_lig, on="ligand", how="inner")
                  .merge(tgt_rec, on="receptor", how="inner")
            )

            if tmp.empty:
                continue

            tmp["slice_id"] = slice_id
            tmp["patient"] = patient
            tmp["source"] = src
            tmp["target"] = tgt
            tmp["n_source"] = n_src
            tmp["n_target"] = n_tgt
            tmp["lr_pair"] = tmp["ligand"] + "→" + tmp["receptor"]

            tmp["comm_score"] = np.sqrt(
                np.maximum(tmp["ligand_mean_source"], 0) *
                np.maximum(tmp["receptor_mean_target"], 0)
            )
            tmp["pct_product"] = tmp["ligand_pct_source"] * tmp["receptor_pct_target"]

            # Directional specificity compared with all non-source / non-target spots.
            non_src_mask = ~src_mask
            non_tgt_mask = ~tgt_mask

            # Only calculate lightweight specificity using already computed genes.
            # These background means help downweight ubiquitous LR expression.
            bg_lig = mean_expr_for_genes(adata, non_src_mask, needed_ligands).rename(columns={
                "gene": "ligand",
                "mean_expr": "ligand_mean_background_source"
            })[["ligand", "ligand_mean_background_source"]]

            bg_rec = mean_expr_for_genes(adata, non_tgt_mask, needed_receptors).rename(columns={
                "gene": "receptor",
                "mean_expr": "receptor_mean_background_target"
            })[["receptor", "receptor_mean_background_target"]]

            tmp = tmp.merge(bg_lig, on="ligand", how="left").merge(bg_rec, on="receptor", how="left")

            eps = 1e-9
            tmp["ligand_log2fc_source"] = np.log2((tmp["ligand_mean_source"] + eps) /
                                                  (tmp["ligand_mean_background_source"] + eps))
            tmp["receptor_log2fc_target"] = np.log2((tmp["receptor_mean_target"] + eps) /
                                                    (tmp["receptor_mean_background_target"] + eps))
            tmp["specificity_score"] = tmp["comm_score"] * np.maximum(tmp["ligand_log2fc_source"], 0) * np.maximum(tmp["receptor_log2fc_target"], 0)

            rows.append(tmp)

    if len(rows) == 0:
        return pd.DataFrame()

    res = pd.concat(rows, ignore_index=True)

    # Ranking: prefer strong expression and target/source specificity.
    # If specificity is all zero, communication score still ranks.
    res["rank_score"] = res["specificity_score"]
    if np.nanmax(res["rank_score"].values) <= 0:
        res["rank_score"] = res["comm_score"] * res["pct_product"]

    res = res.sort_values(
        ["rank_score", "comm_score", "pct_product"],
        ascending=False
    ).reset_index(drop=True)

    return res
